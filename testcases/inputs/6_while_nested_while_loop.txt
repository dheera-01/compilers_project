# calculating the factorial
assign fact = 1;
assign i = 1;
while(i<=5)
{
    fact = fact*i;
    i += 1;
}
#print(i);

# nested while loop
i = 0;
assign j = 0;
assign str = "";
while(i < 3){
    j = 0;
    while(j < 3){
        str = "";
        str = str ~ i ~ " " ~ j;
        print(str);
        j = j+1;
    }
    i = i+1;
}
