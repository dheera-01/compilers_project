# simple for loop
# calculating the factorial
assign fact = 1;
assign i = 1;
# can also use
# for(assign i = 0; i<5; i+= 1)
for(i = 1; i<=5; i = i + 1;)
{
    fact = fact*i;
}
#print(i);
print(fact);

# nested for loop
assign str = "";
i = 0;
for(i = 0; i < 3; i = i + 1;){
    assign j = 0;
    for(j = 0; j<3; j = j+1;){
        str = "";
        str = str ~ i ~ " " ~ j;
        print(str);
    }
}
